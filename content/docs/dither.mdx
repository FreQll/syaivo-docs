## Usage

```ts tab="Vanilla"
import { createDither } from "syaivo";

const effect = createDither({
  colors: ["#c084fc", "#818cf8", "#38bdf8", "#2dd4bf"],
  pixelSize: 4,
  pattern: "bayer",
  angle: 135,
});

effect.mount(document.getElementById("hero")!);
```

```tsx tab="React"
import { Dither } from "syaivo/react";

<Dither
  colors={["#c084fc", "#818cf8", "#38bdf8", "#2dd4bf"]}
  pixelSize={4}
  pattern="bayer"
  angle={135}
  style={{ position: "absolute", inset: 0 }}
/>;
```
